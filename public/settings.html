Рекуррентные платежи активированы.'); closePaymentModal(); // Redirect to dashboard or reload window.location.href = '/dashboard.html'; } else { alert('Ошибка оформления подписки: ' + result.error); } } catch (error) { console.error('[SUBSCRIPTION] Error processing payment success:', error); alert('Ошибка обработки платежа'); } }
